create PROCEDURE [dbo].[CHECK_FREE_ITEM_DATA]
(
	@GID VARCHAR(50),
	@Qty INT
)
AS
BEGIN
	IF(@Qty=(SELECT COUNT(Id) FROM tbl_Cart WHERE GroupID=@GID AND Type='ITEM' AND PriceType='FREE SET'))
	BEGIN
		UPDATE tbl_Cart SET Price='0.00' WHERE Id=(SELECT TOP(1) Id from tbl_Cart where GroupID=@GID AND Type='ITEM' AND PriceType='FREE SET' order by Price asc)
		select 'DONE' MSG
	END
	else
	BEGIN
		DELETE FROM tbl_Cart WHERE GroupID=@GID
		select 'ERROR' MSG
	END
END

GO

CREATE proc [dbo].[FillCart_sp]
  (
  @cartId varchar(150)
   )
  as
  begin
 Select Sum(Price) Price, Qty, GroupID from [dbo].[tbl_Cart]
 where CartId=@CartId  group by GroupID,Qty 
  	 cREATE TABLE #Cart
		(
		
		ProductName nvarchar(max),
		GroupID nvarchar(max)
		
			)
			Insert into #Cart
			Select  ProductName,GroupID from [dbo].[tbl_Cart] 	where CartId=@cartId
 
		 ;with SampleDataR as
						(
							select *, ROW_NUMBER() over (partition by GroupID order by ProductName) rownum
						from #Cart
						) 
						select distinct GroupID,(
						select distinct ProductName
							+ case when s1.rownum = (select MAX(rownum) from SampleDataR 
							where GroupID = s1.GroupID)
							then '' else ' , ' end from SampleDataR s1
						where s1.GroupID = s2.GroupID 
							for xml path(''),type).value('(.)[1]','varchar(max)') Item
						from SampleDataR s2

  end

/****** Object:  UserDefinedFunction [dbo].[SplitString]    Script Date: 29-03-2016 16:13:23 ******/

GO
CREATE FUNCTION [dbo].[SplitString]
(    
     @Input NVARCHAR(MAX),
     @Character CHAR(1)
)
RETURNS @Output TABLE (
     Item NVARCHAR(1000)
)
AS
BEGIN
     DECLARE @StartIndex INT, @EndIndex INT

     SET @StartIndex = 1
     IF SUBSTRING(@Input, LEN(@Input) - 1, LEN(@Input)) <> @Character
     BEGIN
           SET @Input = @Input + @Character
     END

     WHILE CHARINDEX(@Character, @Input) > 0
     BEGIN
           SET @EndIndex = CHARINDEX(@Character, @Input)
          
           INSERT INTO @Output(Item)
           SELECT SUBSTRING(@Input, @StartIndex, @EndIndex - 1)
          
           SET @Input = SUBSTRING(@Input, @EndIndex + 1, LEN(@Input))
     END
     RETURN
END
